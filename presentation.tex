\begin{frame}[t]{SWE-bench Verified Mini Dataset}
\scriptsize
\setlength{\leftmargini}{1.2em}
\textbf{What is in the dataset}
\begin{itemize}\setlength{\itemsep}{0.08cm}
    \item 50 real bug-fix tasks from Django and Sphinx.
    \item For each task: GitHub issue text, starting commit, project version, and setup commit.
    \item Reference solution: gold code patch and test patch.
    \item Evaluation tests: \texttt{FAIL\_TO\_PASS} checks fixed by the patch and \texttt{PASS\_TO\_PASS} regression checks.
\end{itemize}

\vspace{0.05cm}
\textbf{Why it is needed}
\begin{itemize}\setlength{\itemsep}{0.08cm}
    \item Gives realistic repository-level tasks without running the full SWE-bench Verified set.
    \item Keeps evaluation focused on two mature Python codebases, so results are easier to compare and explain.
    \item Provides ground truth patches and tests for measuring whether generated code and explanations match the real fix.
\end{itemize}
\end{frame}

\begin{frame}[t]{Evaluation Plan}
\scriptsize
\begin{enumerate}\setlength{\itemsep}{0.12cm}
    \item Run each Django task twice: baseline agent and agent with explanations.
    \item Check correctness with the dataset test patch and existing regression tests.
    \item Compare solved tasks: \texttt{PASS}/\texttt{FAIL} before and after explanations.
    \item Compare agent efficiency: number of actions on tasks with passed evaluation.
    \item Review explanations for grounding: do they describe the actual code changes?
\end{enumerate}

\textbf{Code generation:} Junie gemini-3-flash-preview
\end{frame}

\begin{frame}[t]{Summary}
\raggedright
\scriptsize
\begin{itemize}\setlength{\itemsep}{0.14cm}
    \item Built a prototype for code generation with aligned explanations.
    \item Prepared a Django-only evaluation setup using SWE-bench Verified Mini.
    \item Initial result: solved tasks stayed at 20/25 after adding explanations.
    \item On passed tasks, explanations reduced Junie's actions in 12/20 cases.
\end{itemize}

\vspace{0.12cm}
\textbf{Next:} evaluate explanation grounding and extend experiments to more repositories and agents.
\end{frame}
